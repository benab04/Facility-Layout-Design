\section{Problem and inputs}

\begin{frame}[t]{Scenario and objectives}
  \RaggedRight
  \small
  \begin{itemize}
    \setlength{\itemsep}{0.35em}
    \item \textbf{Setting:} Five-department cell. Same REL and flow data for every method.
    \item \textbf{Goal:} Two construction layouts (ALDEP, CORELAP), each refined with CRAFT under one transport cost.
    \item \textbf{Cost:} Rectilinear distance between centroids, weighted by flow $F_{ij}$.
    \item \textbf{Output:} Layouts, progress curves, CRAFT cost vs.\ iteration, consistent colors by department.
  \end{itemize}
\end{frame}

\begin{frame}[t]{Departments and areas}
  \centering
  \small
  \begin{tabular}{@{}clc@{}}
    \toprule
    ID & Department & Cells \\
    \midrule
    0 & Machining  & 3 \\
    1 & Assembly   & 2 \\
    2 & Tooling    & 1 \\
    3 & Testing    & 2 \\
    4 & Shipping   & 1 \\
    \bottomrule
  \end{tabular}

  \vspace{0.55em}
  \RaggedRight
  \small
  \begin{itemize}
    \setlength{\itemsep}{0.35em}
    \item One \textbf{REL chart} (A, E, I, O, U, X) drives both construction methods.
    \item ALDEP: compact sweep grid. CORELAP: larger grid, multi-cell departments.
  \end{itemize}
\end{frame}

\begin{frame}[t]{Flow matrix and transport cost}
  \centering
  {\footnotesize\setlength{\arraycolsep}{3pt}%
  \resizebox{0.5\linewidth}{!}{$
  F =
  \begin{bmatrix}
    0 & 12 & 4 & 9 & 2 \\
    3 & 0 & 1 & 14 & 11 \\
    5 & 2 & 0 & 3 & 0 \\
    4 & 6 & 0 & 0 & 8 \\
    2 & 3 & 0 & 2 & 0
  \end{bmatrix}
  $}}

  \vspace{0.45em}
  \RaggedRight
  \small
  \begin{itemize}
    \setlength{\itemsep}{0.35em}
    \item $F_{ij}$: intensity from $i$ to $j$.
    \item $C = \sum_{i\neq j} F_{ij}\, d(\mathbf{c}_i,\mathbf{c}_j)$ with rectilinear $d$.
  \end{itemize}
\end{frame}

\section{Overall pipeline}

\begin{frame}[t]{Pipeline --- construction and centroids (steps 1--3)}
  \RaggedRight
  \footnotesize
  \textbf{Input:} Departments, areas, REL chart, flow $F$, feasible CRAFT pairs (here: all pairs).

  \vspace{0.45em}
  \begin{enumerate}
    \setlength{\itemsep}{0.4em}
    \item \textbf{ALDEP:} REL-based sequence; sweep placement $\rightarrow$ layout $L_A$.
    \item \textbf{CORELAP:} TCR order; PR-driven growth from center $\rightarrow$ layout $L_B$.
    \item \textbf{Centroids:} For each $L$, centroid $\mathbf{c}_i$ = mean of that department's cell centers (same coords as~$d$).
  \end{enumerate}
\end{frame}

\begin{frame}[t]{Pipeline --- CRAFT, visualization, output (steps 4--6)}
  \RaggedRight
  \footnotesize
  \begin{enumerate}
    \setcounter{enumi}{3}
    \setlength{\itemsep}{0.4em}
    \item \textbf{CRAFT (twice):} From centroids of $L_A$ and $L_B$: evaluate all feasible swaps; accept best decrease in~$C$; stop when no improvement (or cycle).
    \item \textbf{Floor replay (drawings only):} Replay accepted swap \emph{order} on a grid copy. Equal areas: swap cells. Unequal: give smaller dept.\ $k$ cells taken from the far side of the larger region; rest to larger.
    \item \textbf{Output:} Refined costs, iteration curves, layouts, geometric centroids on replayed grids.
  \end{enumerate}
\end{frame}

\section{ALDEP}

\begin{frame}[t]{ALDEP --- idea}
  \RaggedRight
  \small
  \begin{itemize}
    \setlength{\itemsep}{0.35em}
    \item \textbf{Sequence:} Random start; extend using REL vs.\ policy level (e.g.\ I).
    \item \textbf{Placement:} Serpentine column sweep; prescribed cell count each dept.
    \item \textbf{Score:} REL-weighted adjacency on grid $\neq$ transport~$C$.

\section{CRAFT}

\begin{frame}[t]{CRAFT --- idea (abstract model)}
  \RaggedRight
  \small
  \begin{itemize}
    \setlength{\itemsep}{0.35em}
    \item Each dept.: centroid of its cells.
    \item \textbf{Move:} Swap centroid coordinate pair of two depts.\ (all pairs here).
    \item \textbf{Greedy:} Best improving swap on $C$; repeat; stop if none (or cycle guard).
    \item Centroid model does not redraw unequal footprints by itself.
  \end{itemize}
\end{frame}

\begin{frame}[t]{Interpreting post-CRAFT floor drawings (1/2)}
  \RaggedRight
  \footnotesize
  \textbf{Goal:} Show cell changes while keeping areas consistent when partners differ in size.
  \begin{itemize}
    \setlength{\itemsep}{0.35em}
    \item \textbf{Equal area:} Full cell exchange.
    \item \textbf{Unequal:} Smaller has $k$ cells. Give it $k$ cells of the larger region farthest from the smaller (Manhattan). Remaining union cells $\rightarrow$ larger.
  \end{itemize}
\end{frame}

\begin{frame}[t]{Interpreting post-CRAFT floor drawings (2/2)}
  \RaggedRight
  \footnotesize
  \begin{itemize}
    \setlength{\itemsep}{0.35em}
    \item Replay swaps in same \emph{order} as CRAFT accepted pairs.
    \item Markers: geometric centroids after replay (plot coords).
    \item Abstract $C$ (curves/bars) stays the centroid model. Re-$C$ on replayed grid may differ---drawings are a companion view.
  \end{itemize}
\end{frame}

\section{Conclusions}

\begin{frame}[t]{Summary}
  \RaggedRight
  \small
  \begin{itemize}
    \setlength{\itemsep}{0.35em}
    \item Shared $F$ and REL enable fair comparison of ALDEP vs.\ CORELAP.
    \item They optimize different surrogates; pre-CRAFT $C$ ranking need not match post-CRAFT.
    \item CRAFT: monotone abstract $C$ from each seed; iteration plots show convergence speed.
    \item Floor drawings use explicit replay rules; read them with---not instead of---abstract $C$ curves.
  \end{itemize}
\end{frame}
